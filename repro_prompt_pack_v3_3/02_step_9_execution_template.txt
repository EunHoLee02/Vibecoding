[authoritative_updates YAML 템플릿]
authoritative_updates:
  frozen_values:
    key: value
  supersedes:
    - old_source -> new_source
  deprecated:
    - deprecated_key_or_rule
  carry_forward_only:
    - key1
    - key2
  do_not_carry_forward:
    - key3
    - key4

stage_status:
  maturity: draft|provisional|minimum_complete|fully_verified
  unresolved_conflicts:
    - none
  missing_p0_items:
    - none
