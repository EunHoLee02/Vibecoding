        code: ErrorCode,
        message: str,

        self.message = message

## 6. Repository

### 6-1. `app/db/repositories/supplement_repository.py`

역할:

* supplement 조회/생성/수정/soft delete/restore

```python
from datetime import datetime
from sqlalchemy import select
from sqlalchemy.orm import Session, selectinload

from app.common.enums import SupplementStatus
from app.db.models.supplement import Supplement


class SupplementRepository:
    def get_list(self, db: Session, user_id, include_deleted: bool = False) -> list[Supplement]:
        stmt = (
            select(Supplement)
            .where(Supplement.user_id == user_id)
            .options(selectinload(Supplement.ingredients))
            .order_by(Supplement.updated_at.desc())
        )

        if not include_deleted:
            stmt = stmt.where(Supplement.deleted_at.is_(None))

        return list(db.scalars(stmt).all())

    def get_by_id(self, db: Session, supplement_id, user_id) -> Supplement | None:
        stmt = (
            select(Supplement)
            .where(
                Supplement.id == supplement_id,
                Supplement.user_id == user_id,
            )
            .options(selectinload(Supplement.ingredients))
        )
        return db.scalar(stmt)

    def create(self, db: Session, supplement: Supplement) -> Supplement:
        db.add(supplement)
        db.flush()
        db.refresh(supplement)
        return supplement

    def soft_delete(self, db: Session, supplement: Supplement) -> Supplement:
        supplement.deleted_at = datetime.utcnow()
        supplement.status = SupplementStatus.DELETED
        db.flush()
        db.refresh(supplement)
        return supplement

    def restore(self, db: Session, supplement: Supplement) -> Supplement:
        supplement.deleted_at = None
        supplement.status = SupplementStatus.ACTIVE
        db.flush()
        db.refresh(supplement)
        return supplement
```

## 7. Service

### 7-1. `app/services/supplement_service.py`

역할:

* 입력 검증 보강
* owner check
* soft delete / restore 정책
* analysis_locked 수정 차단

```python
from fastapi import status

from app.common.enums import SupplementSourceType, SupplementStatus
from app.core.exceptions import AppException, ErrorCode
from app.db.models.supplement import Supplement
from app.db.repositories.supplement_repository import SupplementRepository
from app.db.repositories.supplement_ingredient_repository import SupplementIngredientRepository


class SupplementService:
    def __init__(self) -> None:
        self.supplement_repo = SupplementRepository()
        self.ingredient_repo = SupplementIngredientRepository()

    def list_supplements(self, db, user_id):
        return self.supplement_repo.get_list(db=db, user_id=user_id)

    def get_supplement(self, db, user_id, supplement_id):
        supplement = self.supplement_repo.get_by_id(db=db, supplement_id=supplement_id, user_id=user_id)
        if not supplement:
            raise AppException(
                status_code=status.HTTP_404_NOT_FOUND,
                code=ErrorCode.NOT_FOUND,
                message="영양제를 찾을 수 없습니다.",
            )
        return supplement

    def create_manual_supplement(self, db, user_id, payload):
        supplement = Supplement(
            user_id=user_id,
            source_type=SupplementSourceType.MANUAL,
            product_name=payload.product_name,
            manufacturer=payload.manufacturer,
            serving_basis_type=payload.serving_basis_type,
            daily_serving_count=str(payload.daily_serving_count) if payload.daily_serving_count else None,
            memo=payload.memo,
            status=SupplementStatus.ACTIVE,
        )
        self.supplement_repo.create(db, supplement)
        self.ingredient_repo.replace_all(
            db=db,
            supplement=supplement,
            ingredients_payload=[item.model_dump() for item in payload.ingredient_list],
        )
        db.commit()
        db.refresh(supplement)
        return supplement

    def update_supplement(self, db, user_id, supplement_id, payload):
        supplement = self.get_supplement(db, user_id, supplement_id)

        if supplement.deleted_at is not None:
            raise AppException(
                status_code=status.HTTP_409_CONFLICT,
                code=ErrorCode.CONFLICT,
                message="삭제된 영양제는 수정할 수 없습니다.",
            )

        if supplement.status == SupplementStatus.ANALYSIS_LOCKED:
            raise AppException(
                status_code=status.HTTP_409_CONFLICT,
                code=ErrorCode.CONFLICT,
                message="분석 중인 영양제는 수정할 수 없습니다.",
            )

        if payload.product_name is not None:
            supplement.product_name = payload.product_name
        if payload.manufacturer is not None:
            supplement.manufacturer = payload.manufacturer
        if payload.serving_basis_type is not None:
            supplement.serving_basis_type = payload.serving_basis_type
        if payload.daily_serving_count is not None:
            supplement.daily_serving_count = str(payload.daily_serving_count)
        if payload.memo is not None:
            supplement.memo = payload.memo
        if payload.ingredient_list is not None:
            self.ingredient_repo.replace_all(
                db=db,
                supplement=supplement,
                ingredients_payload=[item.model_dump() for item in payload.ingredient_list],
            )

        db.commit()
        db.refresh(supplement)
        return supplement

    def delete_supplement(self, db, user_id, supplement_id):
        supplement = self.get_supplement(db, user_id, supplement_id)

        if supplement.deleted_at is not None:
            raise AppException(
                status_code=status.HTTP_409_CONFLICT,
                code=ErrorCode.CONFLICT,
                message="이미 삭제된 영양제입니다.",
            )

        if supplement.status == SupplementStatus.ANALYSIS_LOCKED:
            raise AppException(
                status_code=status.HTTP_409_CONFLICT,
                code=ErrorCode.CONFLICT,
                message="분석 중인 영양제는 삭제할 수 없습니다.",
            )

        self.supplement_repo.soft_delete(db, supplement)
        db.commit()
        return supplement

    def restore_supplement(self, db, user_id, supplement_id):
        supplement = self.get_supplement(db, user_id, supplement_id)

        if supplement.deleted_at is None:
            raise AppException(
                status_code=status.HTTP_409_CONFLICT,
                code=ErrorCode.CONFLICT,
                message="삭제된 영양제만 복구할 수 있습니다.",
            )

        self.supplement_repo.restore(db, supplement)
        db.commit()
        db.refresh(supplement)
        return supplement
```

        supplement_id=supplement_id,

@router.delete("/{supplement_id}", status_code=status.HTTP_200_OK)

1. 로그인 사용자 자신의 영양제 목록이 정상 조회된다.
2. soft delete 된 영양제는 기본 목록에서 제외된다.
3. 다른 사용자 영양제는 조회되지 않는다.
4. 인증 없이 요청하면 `401 UNAUTHORIZED`.
5. DB 조회 실패 시 `500 INTERNAL_ERROR` 또는 공통 핸들러 처리.

---

## 10-2. `GET /api/v1/supplements/{supplement_id}`

1. 본인 영양제 상세가 정상 조회된다.
2. ingredient_list가 함께 반환된다.
3. 존재하지 않는 ID면 `404 NOT_FOUND`.
4. 다른 사용자 소유면 `404` 또는 정책상 owner conceal 적용.
5. 인증 없이 요청하면 `401 UNAUTHORIZED`.

---

## 10-3. `POST /api/v1/supplements`

1. 필수 필드와 ingredient_list 1개 이상이면 생성 성공.
2. `ingredient_list=[]`면 validation 실패.
3. `amount_value <= 0`이면 `400 INVALID_INPUT`.
4. `product_name` 공백/빈 문자열이면 `400 INVALID_INPUT`.
5. 생성 시 `source_type=manual`이 강제된다.
6. ingredients row가 함께 생성된다.
7. 인증 없이 요청하면 `401 UNAUTHORIZED`.

---

## 10-4. `PATCH /api/v1/supplements/{supplement_id}`

1. 본인 영양제 일부 필드 수정 성공.
2. ingredient_list 교체 수정 성공.
3. 존재하지 않는 영양제 수정 시 `404 NOT_FOUND`.
4. 삭제된 영양제 수정 시 `409 CONFLICT`.
5. `analysis_locked` 상태 수정 시 `409 CONFLICT`.
6. 다른 사용자 영양제 수정 시 접근 차단.
7. 인증 없이 요청 시 `401 UNAUTHORIZED`.

---

## 10-5. `DELETE /api/v1/supplements/{supplement_id}`

1. soft delete 성공 시 `deleted_at`이 기록된다.
2. 삭제 후 상태가 `deleted`로 바뀐다.
3. 이미 삭제된 항목 재삭제 시 `409 CONFLICT`.
4. `analysis_locked` 상태면 삭제 차단.
5. 다른 사용자 영양제 삭제 차단.
6. 인증 없이 요청 시 `401 UNAUTHORIZED`.

---

## 10-6. `POST /api/v1/supplements/{supplement_id}/restore`

1. 삭제된 영양제 복구 성공.
2. 복구 후 `deleted_at=None`, `status=active`.
3. 삭제되지 않은 항목 복구 시 `409 CONFLICT`.
4. 존재하지 않는 영양제 복구 시 `404 NOT_FOUND`.
5. 다른 사용자 영양제 복구 차단.
6. 인증 없이 요청 시 `401 UNAUTHORIZED`.

---

# 구현 시 주의사항

1. `daily_serving_count`는 현재 예시에서 문자열 저장으로 적었지만, 실제 DB는 `NUMERIC`이 더 적절합니다.
   지금 예시는 구조 설명용이고, 실제 모델 구현에서는 `Numeric(10, 2)` 또는 유사 타입으로 고정하는 것을 권장합니다.
   이 부분은 확실하지 않음이 아니라 **수정 권장 사항**입니다.

2. `SAEnum` 대신 9단계 기준처럼 `VARCHAR + CHECK constraint`를 쓰고 싶다면, ORM 필드 선언을 문자열 기반으로 바꾸는 쪽이 더 9단계와 잘 맞습니다. 지금 코드는 가독성 우선 예시입니다.

3. owner check 시 보안상 다른 사용자 리소스는 `403`보다 `404`로 숨기는 정책이 더 안전할 수 있습니다. 이건 구현 정책 선택 사항입니다.

# 추가 확인 필요 항목

1. `daily_serving_count` DB 타입을 `NUMERIC`로 최종 고정할지
2. enum 저장을 `SAEnum`으로 갈지, 9단계 원칙대로 `VARCHAR + CHECK`로 갈지
3. 다른 사용자 리소스 접근 시 `403` vs `404` 은닉 정책 최종 선택

# 충돌 가능 항목

1. 6단계의 단순 supplement 필드 예시를 그대로 쓰면 분석 preview/analysis 단계와 다시 충돌합니다.
2. OCR confirm 시 supplement 생성 규약이 이번 manual supplement 구조와 동일하게 맞춰져야 합니다.
3. analysis 쪽 fingerprint 계산 시 ingredient ordering 정규화가 빠지면 재사용 판정이 흔들릴 수 있습니다.
